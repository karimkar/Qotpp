// دالة فتح نافذة الحساب الشخصي window.openProfileModal = function() { document.getElementById('profile-modal').classList.remove('hidden'); }; // دالة فتح نافذة المفضلة window.openFavoritesModal = function() { document.getElementById('favorites-modal').classList.remove('hidden'); }; // دالة فتح نافذة البحث window.openSearchModal = function() { document.getElementById('search-modal').classList.remove('hidden'); }; // دالة فتح الصفحة الرئيسية window.openHome = function() { // إغلاق أي نوافذ مفتوحة والعودة للصفحة الرئيسية closeAllModals(); setActiveNavItem('home'); }; // دالة تعيين العنصر النشط في القائمة window.setActiveNavItem = function(itemName) { document.querySelectorAll('.mobile-nav-item').forEach(item => { item.classList.remove('active'); }); // يمكن إضافة منطق لتحديد العنصر النشط حسب itemName if (itemName === 'home') { document.querySelector('.mobile-nav-item:nth-child(1)').classList.add('active'); } }; // دالة إغلاق جميع النوافذ window.closeAllModals = function() { const modals = document.querySelectorAll('.modal-overlay'); modals.forEach(modal => { modal.classList.add('hidden'); }); }; // دالة إغلاق النافذة عند النقر خارجها window.closeModalOnOverlay = function(event, modalId) { if (event.target === document.getElementById(modalId)) { closeModal(modalId); } };
